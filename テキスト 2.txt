```javascript
const CACHE_NAME = 'testa-v1.6-premium-cache';
const ASSETS_TO_CACHE = [
  './index.html',
  './manifest.json',
  'https://cdn.tailwindcss.com',
  'https://unpkg.com/lucide@latest',
  'https://cdn.jsdelivr.net/npm/chart.js'
];

// サービスワーカーがインストールされたときにアセットをキャッシュ
self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => {
      return cache.addAll(ASSETS_TO_CACHE);
    })
  );
  self.skipWaiting(); // 新しいSWが検出されたら待機せずに即座にアクティブにする
});

// 古い古い不要なキャッシュを削除し、最新の修正にアップデート
self.addEventListener('activate', (event) => {
  event.waitUntil(
    caches.keys().then((keys) => {
      return Promise.all(
        keys.map((key) => {
          if (key !== CACHE_NAME) {
            return caches.delete(key);
          }
        })
      );
    })
  );
  self.clients.claim(); // すべてのクライアント（タブ）を即座に新しいSWに制御させる
});

// フェッチ処理：ネットワークファースト（開発効率重視、オンライン時は常に最新を読み込む）
self.addEventListener('fetch', (event) => {
  if (event.request.method !== 'GET') return;
  
  event.respondWith(
    fetch(event.request)
      .then((response) => {
        // レスポンスが正常なら新しいアセットをキャッシュに格納
        if (response.status === 200) {
          const responseClone = response.clone();
          caches.open(CACHE_NAME).then((cache) => {
            cache.put(event.request, responseClone);
          });
        }
        return response;
      })
      .catch(() => {
        // オフライン（電波がない）ときはキャッシュされたデータを返す
        return caches.match(event.request);
      })
  );
});

```
